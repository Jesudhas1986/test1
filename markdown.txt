import os
import markdown

md_dir = "./docs"
for filename in os.listdir(md_dir):
    if filename.endswith(".md"):
        with open(os.path.join(md_dir, filename), "r", encoding="utf-8") as md_file:
            md_text = md_file.read()
        html = markdown.markdown(md_text)
        output_file = os.path.splitext(filename)[0] + ".html"
        with open(os.path.join(md_dir, output_file), "w", encoding="utf-8") as html_file:
            html_file.write(html)

print("All markdown files converted to HTML.")
